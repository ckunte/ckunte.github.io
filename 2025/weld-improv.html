Endurance gain from weld improvements

16 Nov 2025, 3 min

DNV has published RP-C203 Fatigue design of offshore steel structures (2025). It now covers bankable endurance1 from weld improvement techniques such as grinding, and hammer peening.

I’ve taken a stab at comparing Tables 2-3 and F-15, see figure below. The legend to be read as [N] – no improvements, [G] – grinding, and [P] – hammer peening.

#!/usr/bin/env python3
# -*- encoding: utf-8 -*-
# Grinding and hammer peening improvements in DNV-RP-C203 (2025)
# snc.py  -- 2025 C Kunte

# uv:require numpy
# uv:require pandas
# uv:require matplotlib

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import itertools
import hashlib


def lbls():
    """Apply consistent labels, legend, and grid."""
    plt.legend(loc="best", fontsize="small")
    plt.xlabel(r"Number of cycles, N")
    plt.ylabel(r"Hotspot stress, $\sigma$ (MPa)")
    plt.grid(True, which="both", linestyle="--", linewidth=0.5)


def sncurve(
    curve,
    r_start,
    r_mid,
    r_end,
    a1,
    m1,
    a2,
    m2,
    color=None,
    linestyle=None,
):
    # First slope
    n1 = np.logspace(np.log10(r_start), np.log10(r_mid), num=200)
    s1 = (n1 / 10**a1) ** (-1 / m1)

    # Second slope, shifted to connect smoothly
    n2 = np.logspace(np.log10(r_mid), np.log10(r_end), num=200)
    s_mid = (r_mid / 10**a1) ** (-1 / m1)  # endpoint of first slope
    s2 = s_mid * (n2 / r_mid) ** (
        -1 / m2
    )  # scale so it starts from s_mid

    # Plot first segment with label
    plt.loglog(
        n1,
        s1,
        base=10,
        label=f"{curve} (m1={m1}, m2={m2})",
        color=color,
        linestyle=linestyle,
    )

    # Plot second segment without label
    plt.loglog(n2, s2, base=10, color=color, linestyle=linestyle)


def deterministic_color(name, colors):
    """
    Return a deterministic color from a list, based on the curve name.
    Uses a hash of the name so the color assignment is stable across runs.
    """
    idx = int(hashlib.sha256(name.encode()).hexdigest(), 16) % len(
        colors
    )
    return colors[idx]


def main():
    # === Read CSV file ===
    # csv_file = input("Enter CSV filename (e.g., data.csv): ").strip()
    df = pd.read_csv("./sncurves_t.csv")

    # === Define at least 10 distinct line styles ===
    linestyles = [
        "-",
        "--",
        "-.",
        ":",
        (0, (5, 10)),
        (0, (3, 5, 1, 5)),
        (0, (1, 10)),
        (0, (5, 1)),
        (0, (3, 1, 1, 1)),
        (0, (2, 2)),
    ]

    # Use 10 distinct colors from matplotlib's tab10 colormap
    colors = list(plt.cm.tab10.colors)

    # Create a line style cycle
    linestyle_cycle = itertools.cycle(linestyles)

    # === Plot all curves ===
    for _, row in df.iterrows():
        color = deterministic_color(str(row["curve"]), colors)
        linestyle = next(linestyle_cycle)
        sncurve(**row.to_dict(), color=color, linestyle=linestyle)

    lbls()
    plt.title(f"S–N curves as per DNV RP-C203 (2025)")
    plt.savefig("sncurves_t.svg")


if __name__ == "__main__":
    main()
